library(magick)

image_1a <- image_read("https://assets.petco.com/petco/image/upload/f_auto,q_auto/green-tree-frog-care-sheet-hero")

image_2a <- image_blank(width = 500, 
            height = 500, 
            color = "#696969") %>%
  image_annotate(text = "top text",
               color = "#800835",
               size = 40,
               font = "Impact")

image_1b <- image_read("https://food.fnr.sndimg.com/content/dam/images/food/fullset/2008/6/12/0/crab.jpg.rend.hgtvcom.616.462.suffix/1383078047039.jpeg")

image_2b <- image_blank(width = 500, 
                        height = 500, 
                        color = "#696969") %>%
  image_annotate(text = "bottom text",
                 color = "#800835",
                 size = 40,
                 font = "Impact")


images_1 <- c(image_1a,image_2a)
images_2 <- c(image_1b,image_2b)

top_row <- image_append(images_1)
bottom_row <- image_append(images_2)

meme <- c(top_row, bottom_row) %>%
  image_append(stack = TRUE)

image_write(meme, "meme.png")

library(magick)

# image of crab
default_image <- image_read("https://www.nps.gov/asis/planyourvisit/images/Crab-life-cycle-1jpg.jpg")

frame_1 <- default_image %>%
  image_annotate(text = "get rotated",
                 color = "#800835",
                 size = 40,
                 font = "Impact")
frame_2 <- default_image %>% image_rotate(90) %>%
  image_annotate(text = "get rotated",
                 color = "#800835",
                 size = 40,
                 font = "Impact")
frame_3 <- default_image %>% image_rotate(180) %>%
  image_annotate(text = "get rotated",
                 color = "#800835",
                 size = 40,
                 font = "Impact")
frame_4 <- default_image %>% image_rotate(270) %>%
  image_annotate(text = "get rotated",
                 color = "#800835",
                 size = 40,
                 font = "Impact")

# rotation
frames <- c(frame_1,frame_2,frame_3,frame_4)
animation <- image_animate(frames, fps = 5)

image_write(animation, "crab_cycle.gif")
